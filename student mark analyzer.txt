def analyze_student_marks(marks_list):
    # 1. Remove invalid marks (less than 0 or greater than 100)
    valid_marks = [mark for mark in marks_list if 0 <= mark <= 100]
    
    if not valid_marks:
        return "No valid marks to analyze."

    # 2. Calculate average
    average = sum(valid_marks) / len(valid_marks)

    # 3. Find topper(s)
    top_mark = max(valid_marks)
    toppers = [mark for mark in valid_marks if mark == top_mark]

    # 4. Display grades based on average
    if average >= 90:
        grade = 'A'
    elif average >= 80:
        grade = 'B'
    elif average >= 70:
        grade = 'C'
    elif average >= 60:
        grade = 'D'
    else:
        grade = 'F'

    # Output Results
    print(f"Valid Marks: {valid_marks}")
    print(f"Average Mark: {average:.2f}")
    print(f"Topper Mark: {top_mark} (Found {len(toppers)} topper(s))")
    print(f"Overall Grade: {grade}")

# Example Usage:
student_marks = [85, 92, 105, -5, 78, 92, 60]
analyze_student_marks(student_marks)
