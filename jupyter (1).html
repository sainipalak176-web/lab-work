# Product names
products = ['Laptop', 'Mobile', 'Tablet', 'Headphones', 'Camera', 'Printer']

# Sales data
sales = [50, 80, 40, 60, 30, 45]

# Different colors for each bar
colors = ['red', 'blue', 'green', 'orange', 'purple', 'cyan']

# Labels and title

plt.ylabel("Sales")
plt.title("Sales of 6 Products")
