import matplotlib.pyplot as plt

# Data
x = [1, 2, 3, 4, 5]
y = [10, 20, 30, 40, 50]

categories = ['A', 'B', 'C', 'D']
values = [15, 25, 35, 20]

pie_labels = ['Apple', 'Banana', 'Mango', 'Orange']
pie_sizes = [30, 25, 25, 20]

hours = [1, 2, 3, 4, 5]
marks = [35, 40, 50, 60, 70]

# Create subplots
plt.figure(figsize=(10,8))

# Line Chart
plt.subplot(2,2,1)
plt.plot(x, y)
plt.title("Line Chart")

# Bar Chart
plt.subplot(2,2,2)
plt.bar(categories, values)
plt.title("Bar Chart")

# Pie Chart
plt.subplot(2,2,3)
plt.pie(pie_sizes, labels=pie_labels, autopct='%1.1f%%')
plt.title("Pie Chart")

# Scatter Plot
plt.subplot(2,2,4)
plt.scatter(hours, marks)
plt.title("Scatter Plot")

plt.tight_layout()
plt.show()
