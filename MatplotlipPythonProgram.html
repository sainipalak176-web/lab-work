import matplotlib.pyplot as plt

# Cities
cities = ["Delhi", "Mumbai", "Kolkata", "Chennai", "Bangalore"]

# Data
population = [19, 20, 14, 11, 13]   # in millions
area = [1484, 603, 185, 426, 709]   # in km²
gdp = [293, 310, 150, 160, 180]     # GDP values (for bubble size)

# Create bubble chart
plt.scatter(area, population, s=[g*10 for g in gdp], alpha=0.5)

# Labels for each city
for i, city in enumerate(cities):
    plt.text(area[i], population[i], city)

# Titles and labels
plt.title("Population vs Area of Cities (Bubble Size = GDP)")
plt.xlabel("Area (km²)")
plt.ylabel("Population (millions)")

# Display chart
plt.show()

import matplotlib.pyplot as plt

# Data
x = [1, 2, 3, 4, 5]
y = [10, 20, 25, 30, 40]

courses = ["Python", "Java", "C++"]
students = [40, 35, 25]

brands = ["Samsung", "Apple", "Xiaomi"]
share = [40, 35, 25]

hours = [1, 2, 3, 4, 5]
marks = [30, 40, 50, 65, 80]

# Create subplots (2 rows, 2 columns)
plt.figure(figsize=(8,6))

# Line Chart
plt.subplot(2,2,1)
plt.plot(x, y, marker='o')
plt.title("Line Chart")

# Bar Chart
plt.subplot(2,2,2)
plt.bar(courses, students)
plt.title("Bar Chart")

# Pie Chart
plt.subplot(2,2,3)
plt.pie(share, labels=brands, autopct='%1.1f%%')
plt.title("Pie Chart")

# Scatter Plot
plt.subplot(2,2,4)
plt.scatter(hours, marks)
plt.title("Scatter Plot")

plt.tight_layout()
plt.show()
